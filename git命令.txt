1  git  init				创建版本库，在项目文件夹内，执行；
2  git  status  			进行查看文件状态   
3  git add <文件名>			将文件添加到暂存区    
4  git commit Cm “注释内容”		提交文件到本地库,带注释提交
5  git log <文件名>			查看历史记录
6  git log --pretty=online <文件名> 	简易查看信息
7  git reset --hard HEAD^		回退上一次提交
8  git reset --hard HEAD~n		回退n次操作
9  git reflog <文件名>			查看历史记录版本号
10 git reset -hard <版本号>		穿越到某个版本
11 git checkout --<文件名> 		还原文件
12 git stash				先将本地修改暂存起来
13 git stash pop 			还原暂存内容
14 git branch <分支名>			创建分支
15 git branch -v			查看分支
16 git checkout <分支名>		切换分支
17 git checkout -b <分支名>		新建分支并切换到该分支
18 git merge <分支名> 			(先切换到主分支)合并分支
19 git branch -D <分支名> 		(先切换到主分支)删除分支
20 git diff 				可以找到冲突的文件和冲突的内容
21 git clone <版本库网址> <本地库名>	从远程克隆一个版本库，本地库名不写默认和远程相同
22 git clone -o <版本库网址> <本地库名>	克隆时，-o选项自定义主机名

# --------remote相关--------
23 git remote 				列出所有远程主机名称
24 git remote -v 			查看远程主机网址
25 git remote show <主机名>		查看主机详细信息
26 git remote add <主机名> <网址>	添加远程主机
27 git remote rm <主机名> 		删除远程主机
28 git remote rename <原主机名> <新主机名> 主机重命名


29 git fetch <远程主机名>		将远端更细取回本地。用来查看他人进程，与本地开发代码没有影响
30 git fetch <远程主机名> <分支名>	取回远程指定分支的更新，默认取回所有分支的更新
31 git branch -r 			查看远程分支
32 git branch -a 			查看所有分支
33 git checkout -b <newBranch> origin/master 在远程创建一个新分支
34 git merge origin/master 		在本地当前分支上合并远程分支origin/master
35 git rebase origin/master		在本地当前分支上合并远程分支origin/master

#--------git pull相关--------
36 git pull <远程主机名> <远程分支名>:<本地分支名>  取回远程主机分支的更新再与本地指定分支合并;		git pull = git fetch + git merge
37 git pull <远程主机名> <远程分支名>	与本地当前分支合并
38 git pull <远程主机名> 		如果当前分支与远程分支存在追踪关系，可省略远程分支
39 git pull				当前分支自动与唯一一个追踪分支进行更新合并
40 git branch --set-upstream master origin/next 手动将本地master和远程next分支建立追踪关系
41 git pull --rebase <远程主机名> <远程分支名>:<本地分支名> 使用rebase方式合并
###如果远程主机删除了某个分支，git pull默认情况下不会在拉取远程分支的时候删除本地分支。如果想改变这个默认行为，使用git pull -p

#--------git push相关--------
42 git push <远程主机名> <本地分支名>:<远程分支名> 将本地分支的更新推送到远程分支； #注意，分支推送的写法是<来源地>:<目的地>，所以git pull是<远程分支>:<本地分支>，git push是<本地分支>:<远程分支>
43 git push <远程主机名> <本地分支名>  	省略远程分支名，默认推送到与之存在追踪关系的远程分支，如果远程分支不存在，则会被新建
44 git push <远程主机名>  :<远程分支>	如果省略了本地分支名，相当于删除远程分支，因为这相当于推送了一个空的本地分支到远程分支； 等同于git push origin --delete <远程分支名>
45 git push <远程主机名>		如果当前分支与远程分支存在追踪关系，则都可以省略
46 git push 				如果当前分支只有一个追踪分支，那么主机名都可以省略
47 git push -u <远程主机名> <本地分支名> 将本地的master分支推送到origin主机，同时指定origin为默认主机，后面就可以不加任何参数使用git push了；
# 不带任何参数的git push，默认只推送当前分支，这叫做simple方式。此外，还有一种matching方式，会推送所有有对应的远程分支的本地分支。Git 2.0版本之前，默认采用matching方法，现在改为默认采用simple方式。如果要修改这个设置，可以采用git config命令:
git config --global push.default matching
# 或者
git config --global push.default simple 
# 还有一种情况，就是不管是否存在对应的远程分支，将本地的所有分支都推送到远程主机，这时需要使用--all选项
git push --all origin
# 如果远程主机的版本比本地版本更新，推送时Git会报错，要求先在本地做git pull合并差异，然后再推送到远程主机。这时，如果你一定要推送，可以使用--force选项;
git push --force origin 
#上面命令使用--force选项，结果导致远程主机上更新的版本被覆盖。除非你很确定要这样做，否则应该尽量避免使用--force选项;

# 最后，git push不会推送标签（tag），除非使用--tags选项。
$ git push origin --tags
